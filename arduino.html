Introducción a Arduino

Arduino es una plataforma de hardware y software de código abierto que permite crear proyectos interactivos mediante la programación de microcontroladores. Esta guía te proporcionará los conocimientos esenciales para programar en Arduino.

Estructura Básica de un Sketch

// Sección de definiciones y includes
#include <biblioteca.h>
#define CONSTANTE valor

// Declaración de variables globales
tipoDato variable = valor;

// Configuración inicial (se ejecuta una vez)
void setup() {
  // Inicializar pines, comunicaciones, etc.
}

// Bucle principal (se ejecuta repetidamente)
void loop() {
  // Código principal del programa
}

Funciones Esenciales de Arduino

1. Configuración de Pines

pinMode(pin, mode) - Configura un pin específico como entrada o salida.

void setup() {
  pinMode(13, OUTPUT); // Pin 13 como salida
  pinMode(A0, INPUT); // Pin analógico A0 como entrada
  pinMode(2, INPUT_PULLUP); // Pin 2 como entrada con resistencia pull-up interna
}

2. Control de Tiempo

delay(ms) - Pausa el programa durante el tiempo especificado en milisegundos.

void loop() {
  digitalWrite(13, HIGH);
  delay(500); // Espera 500 milisegundos
  digitalWrite(13, LOW);
  delay(500);
}

3. Comunicación Serial

Serial.begin(velocidad) - Inicializa la comunicación serial.

void setup() {
  Serial.begin(9600); // 9600 baudios
}

void loop() {
  int valor = analogRead(A0);
  Serial.print("Valor del sensor: ");
  Serial.println(valor);
  delay(1000);
}

4. Funciones Matemáticas

map(valor, fromLow, fromHigh, toLow, toHigh) - Re-mapea un número de un rango a otro.

void loop() {
  int valorSensor = analogRead(A0);
  int valorMapeado = map(valorSensor, 0, 1023, 0, 255);
  analogWrite(9, valorMapeado);
}

Ejemplo Completo

// Control de LED con sensor de luz y comunicación serial

#define LED_PIN 13
#define LDR_PIN A0

int umbralLuz = 500;
bool estadoLED = false;

void setup() {
  pinMode(LED_PIN, OUTPUT);
  Serial.begin(9600);
  Serial.println("Sistema iniciado");
}

void loop() {
  int valorLuz = analogRead(LDR_PIN);
  
  if (valorLuz < umbralLuz && !estadoLED) {
    encenderLED();
  } else if (valorLuz >= umbralLuz && estadoLED) {
    apagarLED();
  }
  
  if (Serial.available() > 0) {
    procesarComandoSerial();
  }
  
  delay(100);
}

void encenderLED() {
  digitalWrite(LED_PIN, HIGH);
  estadoLED = true;
  Serial.println("LED encendido");
}

void apagarLED() {
  digitalWrite(LED_PIN, LOW);
  estadoLED = false;
  Serial.println("LED apagado");
}

void procesarComandoSerial() {
  char comando = Serial.read();
  if (comando == 'E') {
    encenderLED();
  } else if (comando == 'A') {
    apagarLED();
  } else if (comando == 'U') {
    umbralLuz = Serial.parseInt();
    Serial.print("Umbral cambiado a: ");
    Serial.println(umbralLuz);
  }
}
